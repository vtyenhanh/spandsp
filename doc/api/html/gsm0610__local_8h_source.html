38 static __inline__ int16_t gsm_add(int16_t a, int16_t b)
39 {
40 #if defined(__GNUC__) && (defined(__i386__) || defined(__x86_64__))
41  __asm__ __volatile__(
42  " addw %2,%0;\n"
43  " jno 0f;\n"
44  " movw $0x7fff,%0;\n"
45  " adcw $0,%0;\n"
46  "0:"
47  : "=&r" (a)
48  : "0" (a), "ir" (b)
49  : "cc"
50  );
51  return a;
52 #else
53  int32_t sum;

65  " addl %2,%0;\n"

72  : "cc"
73  );
74  return a;

141 {
