32 /*!
33  V.27ter modem transmit side descriptor. This defines the working state for a

36 struct v27ter_tx_state_s

43  void *get_bit_user_data;
44 
45  /*! \brief The callback function used to report modem status changes. */
46  span_modem_status_func_t status_handler;
47  /*! \brief A user specified opaque pointer passed to the status function. */
48  void *status_user_data;
49 
50 #if defined(SPANDSP_USE_FIXED_POINT)
51  /*! \brief The gain factor needed to achieve the specified output power at 2400bps. */
52  int16_t gain_2400;
53  /*! \brief The gain factor needed to achieve the specified output power at 4800bps. */
54  int16_t gain_4800;
55  /*! \brief The root raised cosine (RRC) pulse shaping filter buffer. */
56  int16_t rrc_filter_re[V27TER_TX_FILTER_STEPS];
57  int16_t rrc_filter_im[V27TER_TX_FILTER_STEPS];
58 #else
59  /*! \brief The gain factor needed to achieve the specified output power at 2400bps. */
60  float gain_2400;
61  /*! \brief The gain factor needed to achieve the specified output power at 4800bps. */
62  float gain_4800;
63  /*! \brief The root raised cosine (RRC) pulse shaping filter buffer. */
64  float rrc_filter_re[V27TER_TX_FILTER_STEPS];
65  float rrc_filter_im[V27TER_TX_FILTER_STEPS];
66 #endif
67 
68  /*! \brief Current offset into the RRC pulse shaping filter buffer. */
69  int rrc_filter_step;
70 
71  /*! \brief The register for the training and data scrambler. */
72  uint32_t scramble_reg;
73  /*! \brief A counter for the number of consecutive bits of repeating pattern through
74  the scrambler. */
75  int scrambler_pattern_count;
76  /*! \brief True if transmitting the training sequence, or shutting down transmission.
77  False if transmitting user data. */
78  bool in_training;
79  /*! \brief A counter used to track progress through sending the training sequence. */
80  int training_step;
81 
82  /*! \brief The current phase of the carrier (i.e. the DDS parameter). */
83  uint32_t carrier_phase;

88  /*! \brief The code number for the current position in the constellation. */

90  /*! \brief The get_bit function in use at any instant. */
91  span_get_bit_func_t current_get_bit;

93  logging_state_t logging;
