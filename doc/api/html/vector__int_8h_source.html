167 static __inline__ int vec_max_bitsi16(const int16_t *vec, int len)
168 {
169  int i;
170  int max;
171  int v;
172  int b;
173 
174  max = 0;
175  for (i = 0; i < len; i++)
176  {
177  v = abs(vec[i]);
178  if (v > max)
179  max = v;
180  }
181  b = 0;
182  while (max != 0)
183  {
184  b++;
185  max >>= 1;
186  }
187  return b;
188 }
