71 static __inline__ int32_t pow_ii(int32_t x, int32_t n)
72 {
73  int32_t pow;
74  uint32_t u;
75 
76  if (n <= 0)
77  {
78  if (n == 0 || x == 1)
79  return 1;
80  if (x != -1)
81  return (x != 0) ? 1/x : 0;
82  n = -n;
83  }
84  u = n;
85  for (pow = 1; ; )
86  {
87  if ((u & 1))
88  pow *= x;
89  if ((u >>= 1) == 0)
90  break;
91  x *= x;
92  }
93  return pow;
94 }
