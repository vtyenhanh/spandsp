68  int consecutive_eols;

74  (See T.4/4.2.1.3.2.). */
75  int a0;

78  int b1;

82  int black_white;

84  bool in_black;
85

87  int a_cursor;

89  int b_cursor;
90 
91  /*! \brief Incoming bit buffer for decompression. */
92  uint32_t rx_bitstream;
93  /*! \brief The number of bits currently in rx_bitstream. */
94  int rx_bits;
95  /*! \brief The number of bits to be skipped before trying to match the next code word. */
96  int rx_skip_bits;
97 
98  /*! \brief Decoded pixel stream buffer. */
99  uint32_t pixel_stream;
100  /*! \brief The number of pixels currently in pixel_stream. */
101  int pixels;
102 
103  /*! \brief The minimum bits in any row of the current page. For monitoring only. */
104  int min_row_bits;
105  /*! \brief The maximum bits in any row of the current page. For monitoring only. */
106  int max_row_bits;
107 
108  /*! \brief The size of the compressed image, in bits. */
109  int compressed_image_size;
110  /*! \brief The current number of consecutive bad rows. */
111  int curr_bad_row_run;
112  /*! \brief The longest run of consecutive bad rows seen in the current page. */

114  /*! \brief The total number of bad rows in the current page. */
